<script>
(function(){
    const canvas = document.getElementById('tetrisCanvas');
    const ctx = canvas.getContext('2d');
    const scoreSpan = document.getElementById('tetrisScore');
    const highScoreSpan = document.getElementById('tetrisHighScore');
    
    const SHAPES = [
        { shape: [[1,1,1,1]], color: '#00ffff' },
        { shape: [[1,1],[1,1]], color: '#ffff00' },
        { shape: [[0,1,0],[1,1,1]], color: '#aa66ff' },
        { shape: [[0,1,1],[1,1,0]], color: '#66ff66' },
        { shape: [[1,1,0],[0,1,1]], color: '#ff6666' },
        { shape: [[1,0,0],[1,1,1]], color: '#ffaa44' },
        { shape: [[0,0,1],[1,1,1]], color: '#4488ff' }
    ];
    
    const COLS = 12;
    const ROWS = 12;
    const CELL_SIZE = canvas.width / COLS;
    
    let board = Array(ROWS).fill().map(() => Array(COLS).fill(0));
    let currentPiece = null;
    let score = 0;
    let highScore = localStorage.getItem('tetrisHighScore') || 0;
    let gameLoop = null;
    let gameRunning = true;
    
    highScoreSpan.textContent = highScore;
    
    function randomPiece() {
        const idx = Math.floor(Math.random() * SHAPES.length);
        const shape = SHAPES[idx].shape.map(row => [...row]);
        return { shape: shape, color: SHAPES[idx].color };
    }
    
    let nextPiece = randomPiece();
    
    function spawnNewPiece() {
        currentPiece = {
            shape: nextPiece.shape.map(row => [...row]),
            color: nextPiece.color,
            x: Math.floor((COLS - nextPiece.shape[0].length) / 2),
            y: 0
        };
        nextPiece = randomPiece();
        if (collision()) {
            gameRunning = false;
            if (gameLoop) clearInterval(gameLoop);
            draw();
            return false;
        }
        draw();
        return true;
    }
    
    function collision() {
        for (let y = 0; y < currentPiece.shape.length; y++) {
            for (let x = 0; x < currentPiece.shape[y].length; x++) {
                if (currentPiece.shape[y][x]) {
                    const boardX = currentPiece.x + x;
                    const boardY = currentPiece.y + y;
                    if (boardX < 0 || boardX >= COLS || boardY >= ROWS) return true;
                    if (boardY >= 0 && board[boardY][boardX]) return true;
                }
            }
        }
        return false;
    }
    
    function mergePiece() {
        for (let y = 0; y < currentPiece.shape.length; y++) {
            for (let x = 0; x < currentPiece.shape[y].length; x++) {
                if (currentPiece.shape[y][x]) {
                    const boardX = currentPiece.x + x;
                    const boardY = currentPiece.y + y;
                    if (boardY >= 0 && boardY < ROWS && boardX >= 0 && boardX < COLS) {
                        board[boardY][boardX] = currentPiece.color;
                    }
                }
            }
        }
        clearLines();
        spawnNewPiece();
        draw();
    }
    
    function clearLines() {
        let linesCleared = 0;
        for (let y = ROWS - 1; y >= 0; y--) {
            let full = true;
            for (let x = 0; x < COLS; x++) {
                if (!board[y][x]) { full = false; break; }
            }
            if (full) {
                board.splice(y, 1);
                board.unshift(Array(COLS).fill(0));
                y++;
                linesCleared++;
            }
        }
        if (linesCleared > 0) {
            score += linesCleared * 50;
            scoreSpan.textContent = score;
            if (score > highScore) {
                highScore = score;
                highScoreSpan.textContent = highScore;
                localStorage.setItem('tetrisHighScore', highScore);
            }
        }
    }
    
    function moveLeft() { if (!gameRunning) return; currentPiece.x--; if (collision()) currentPiece.x++; else draw(); }
    function moveRight() { if (!gameRunning) return; currentPiece.x++; if (collision()) currentPiece.x--; else draw(); }
    function moveDown() { if (!gameRunning) return; currentPiece.y++; if (collision()) { currentPiece.y--; mergePiece(); } draw(); }
    function rotatePiece() {
        if (!gameRunning) return;
        const rotated = currentPiece.shape[0].map((_, idx) => currentPiece.shape.map(row => row[idx]).reverse());
        const original = currentPiece.shape;
        currentPiece.shape = rotated;
        if (collision()) currentPiece.shape = original;
        draw();
    }
    function dropPiece() { if (!gameRunning) return; while (!collision()) currentPiece.y++; currentPiece.y--; mergePiece(); draw(); }
    
    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        for (let y = 0; y < ROWS; y++) {
            for (let x = 0; x < COLS; x++) {
                if (board[y][x]) {
                    ctx.fillStyle = board[y][x];
                    ctx.fillRect(x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE - 0.5, CELL_SIZE - 0.5);
                }
            }
        }
        
        if (currentPiece) {
            for (let y = 0; y < currentPiece.shape.length; y++) {
                for (let x = 0; x < currentPiece.shape[y].length; x++) {
                    if (currentPiece.shape[y][x]) {
                        ctx.fillStyle = currentPiece.color;
                        const boardX = (currentPiece.x + x) * CELL_SIZE;
                        const boardY = (currentPiece.y + y) * CELL_SIZE;
                        ctx.fillRect(boardX, boardY, CELL_SIZE - 0.5, CELL_SIZE - 0.5);
                    }
                }
            }
        }
        
        // 网格线
        ctx.strokeStyle = 'rgba(255,255,255,0.08)';
        for (let x = 0; x <= COLS; x++) {
            ctx.beginPath();
            ctx.moveTo(x * CELL_SIZE, 0);
            ctx.lineTo(x * CELL_SIZE, canvas.height);
            ctx.stroke();
        }
        for (let y = 0; y <= ROWS; y++) {
            ctx.beginPath();
            ctx.moveTo(0, y * CELL_SIZE);
            ctx.lineTo(canvas.width, y * CELL_SIZE);
            ctx.stroke();
        }
        
        if (!gameRunning) {
            ctx.fillStyle = 'rgba(0,0,0,0.7)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = '#ff8888';
            ctx.font = 'bold 10px monospace';
            ctx.textAlign = 'center';
            ctx.fillText('GAME OVER', canvas.width/2, canvas.height/2);
        }
    }
    
    function restart() {
        board = Array(ROWS).fill().map(() => Array(COLS).fill(0));
        score = 0;
        scoreSpan.textContent = score;
        gameRunning = true;
        nextPiece = randomPiece();
        spawnNewPiece();
        if (gameLoop) clearInterval(gameLoop);
        gameLoop = setInterval(() => { if (gameRunning) moveDown(); }, 400);
    }
    
    document.addEventListener('keydown', (e) => {
        if (e.key === 'ArrowLeft') { moveLeft(); e.preventDefault(); }
        else if (e.key === 'ArrowRight') { moveRight(); e.preventDefault(); }
        else if (e.key === 'ArrowDown') { moveDown(); e.preventDefault(); }
        else if (e.key === 'ArrowUp') { rotatePiece(); e.preventDefault(); }
        else if (e.key === ' ') { dropPiece(); e.preventDefault(); }
    });
    
    document.getElementById('tetrisLeft').onclick = () => moveLeft();
    document.getElementById('tetrisRight').onclick = () => moveRight();
    document.getElementById('tetrisRotate').onclick = () => rotatePiece();
    document.getElementById('tetrisDown').onclick = () => moveDown();
    document.getElementById('tetrisDrop').onclick = () => dropPiece();
    document.getElementById('tetrisRestart').onclick = () => restart();
    
    restart();
})();
</script>
