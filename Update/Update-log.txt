<div style="text-align: center;">
    <canvas id="coinCanvas" width="350" height="350" style="background: linear-gradient(135deg, #2d1b4e, #1a0f2e); border-radius: 24px; display: block; margin: 0 auto; box-shadow: 0 8px 32px rgba(0,0,0,0.3); cursor: pointer;"></canvas>
    <div style="margin-top: 12px;">
        <span style="color: #ffd700;">💰 金币: <span id="coinScore" style="font-size: 24px; font-weight: bold; color: #ffd700;">0</span></span>
        <button id="coinRestart" style="background: rgba(255,215,0,0.3); border: none; border-radius: 20px; padding: 6px 20px; color: white; margin-left: 20px; cursor: pointer;">🔄 重来</button>
    </div>
    <p style="font-size: 12px; color: rgba(255,255,255,0.5); margin-top: 8px;">✨ 点击金币淘金币！ ✨</p>

<script>
(function(){
    const canvas = document.getElementById('coinCanvas');
    if(!canvas) return;
    const ctx = canvas.getContext('2d');
    const scoreSpan = document.getElementById('coinScore');
    const restartBtn = document.getElementById('coinRestart');
    
    let coins = [];
    let score = 0;
    let animationId = null;
    let spawnInterval = null;
    let gameRunning = true;
    
    const width = canvas.width = 350;
    const height = canvas.height = 350;
    
    class Coin {
        constructor() {
            this.radius = 15;
            this.x = Math.random() * (width - 2 * this.radius) + this.radius;
            this.y = Math.random() * (height - 2 * this.radius) + this.radius;
            this.value = 1;
        }
        
        draw() {
            ctx.save();
            ctx.shadowBlur = 8;
            ctx.shadowColor = '#ffd700';
            // 金币渐变
            const grad = ctx.createLinearGradient(this.x - 5, this.y - 5, this.x + 5, this.y + 5);
            grad.addColorStop(0, '#ffd700');
            grad.addColorStop(1, '#ffaa00');
            ctx.fillStyle = grad;
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.fill();
            // 内圈
            ctx.fillStyle = '#ffcc00';
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.radius - 3, 0, Math.PI * 2);
            ctx.fill();
            // 钱币符号
            ctx.fillStyle = '#b8860b';
            ctx.font = `bold ${this.radius + 4}px monospace`;
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            ctx.fillText('¥', this.x, this.y);
            ctx.restore();
        }
        
        contains(px, py) {
            const dx = px - this.x;
            const dy = py - this.y;
            return Math.sqrt(dx * dx + dy * dy) <= this.radius;
        }
    }
    
    function initCoins(count = 8) {
        coins = [];
        for(let i = 0; i < count; i++) {
            coins.push(new Coin());
        }
    }
    
    function draw() {
        ctx.clearRect(0, 0, width, height);
        
        // 绘制网格装饰
        ctx.strokeStyle = 'rgba(255,215,0,0.1)';
        for(let i = 0; i < width; i += 30) {
            ctx.beginPath();
            ctx.moveTo(i, 0);
            ctx.lineTo(i, height);
            ctx.stroke();
            ctx.beginPath();
            ctx.moveTo(0, i);
            ctx.lineTo(width, i);
            ctx.stroke();
        }
        
        for(let coin of coins) {
            coin.draw();
        }
    }
    
    function handleClick(e) {
        if(!gameRunning) return;
        const rect = canvas.getBoundingClientRect();
        const scaleX = canvas.width / rect.width;
        const scaleY = canvas.height / rect.height;
        
        let clientX, clientY;
        if(e.touches) {
            clientX = e.touches[0].clientX;
            clientY = e.touches[0].clientY;
            e.preventDefault();
        } else {
            clientX = e.clientX;
            clientY = e.clientY;
        }
        
        const canvasX = (clientX - rect.left) * scaleX;
        const canvasY = (clientY - rect.top) * scaleY;
        
        for(let i = coins.length - 1; i >= 0; i--) {
            if(coins[i].contains(canvasX, canvasY)) {
                coins.splice(i, 1);
                score++;
                if(scoreSpan) scoreSpan.textContent = score;
                
                // 添加新金币
                coins.push(new Coin());
                
                // 特效：短暂闪烁
                canvas.style.filter = 'brightness(1.3)';
                setTimeout(() => { canvas.style.filter = ''; }, 100);
                break;
            }
        }
        draw();
    }
    
    function restart() {
        score = 0;
        if(scoreSpan) scoreSpan.textContent = '0';
        initCoins(8);
        draw();
        gameRunning = true;
    }
    
    canvas.addEventListener('click', handleClick);
    canvas.addEventListener('touchstart', handleClick);
    if(restartBtn) restartBtn.onclick = () => restart();
    
    initCoins(8);
    draw();
    
    // 动画循环（让金币微微跳动）
    let time = 0;
    function animate() {
        if(!gameRunning) return;
        time += 0.05;
        draw();
        requestAnimationFrame(animate);
    }
    animate();
})();
</script>
